import Plots
begin
    x = 0:0.01:2pi
    y = sin.(x)
    z = cos.(x) 
    Plots.plot(x,y,lw=2,lc=:red,ls=:dash,label="sin(x)") 
    Plots.plot!(x,z,lw=1,lc=:blue,ls=:dot,label="cos(x)") 
    Plots.xlabel!("x")
    Plots.ylabel!("y")
end

Built with Julia 1.10.3 and

Plots 1.40.4

To run this tutorial locally, download this file and open it with Pluto.jl.
